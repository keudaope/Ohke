C01. Tee Henkiloluokka, jossa on muuttujat: Etu-, sukunimi ja ikä ja tee konstructori, jonka avulla ko. tiedot voi suoraan syöttää. Tee tämän jälkeen risto-olio seuraavilla tiedoilla: Etunimi: Risto, Sukunimi: Reipas, Ikä: 10 ja tulosta sen jälkeen riston tiedot konsoliin

C#

             using System;
              class Henkilo
              {
                  public string EtuNimi { get; set; }
                  public string SukuNimi { get; set; }
                  public int Ika { get; set; }
                  public Henkilo(string enimi, string snimi, int age)
                  {
                      EtuNimi = enimi;
                      SukuNimi = snimi;
                      Ika = age;
                  }
              }

              class Program
              {
                  static void Main(string[] args)
                  {
                      Henkilo risto = new Henkilo("Risto", "Reipas", 10);
                      Console.WriteLine("Etunimi: " + risto.EtuNimi);
                      Console.WriteLine("Sukunimi: " + risto.SukuNimi);
                      Console.WriteLine("Ikä: " + risto.Ika);
                  }
              }

C04. Tee kuvakirja -niminen luokka, jossa privaatti muuttuja SivujenLukumäärä ja oletuskonsturi, jossa sivumääräksi määritellään 16 sivua sekä konstruktori, jolla voi antaa itse sivumäärän

C#

              using System;
              class Kuvakirja
              {
                  private int sivujenLukumaara;

                  public Kuvakirja()
                  {
                      sivujenLukumaara = 16;
                  }

                  public Kuvakirja(int sivut)
                  {
                      sivujenLukumaara = sivut;
                  }

                  public int SivujenLukumaara
                  {
                      get { return sivujenLukumaara; }
                  }
              }

              class Program
              {
                  static void Main(string[] args)
                  {
                      //luodaan olio oletuskonstruktorin avulla
                      var minunKuvat = new Kuvakirja();
                      Console.WriteLine(minunKuvat.SivujenLukumaara); //tulostaa 16
                      
                      //luodaan olio ylikuormitetulla konstruktorilla
                      var minunToisetKuvat = new Kuvakirja(32);
                      Console.WriteLine(minunToisetKuvat.SivujenLukumaara); //tulostaas 32
                  }
              }

C06. Tee luokka ajoneuvo, jossa on kolme muuttujaa: merkki, malli, vuosimalli. Tee konstruktorit, getterit ja setterit luokkaan.Lisää metodit aja, kiihdyta, jarruta, kaynnista ja sammuta. Tee sitten luokka auto, joka perii luokan ajoneuvo. Tee sitten luokka TestiAjo, jossa luot uuden olion nimeltä mersu ja kutsut metodia aja tuolle mersu oliolle.

C#

              using System;
              class Ajoneuvo
              {
                  private string merkki;
                  private string malli;
                  private int vuosimalli;
                  
                  public Ajoneuvo(string merkki, string malli, int vuosimalli)
                  {
                      this.merkki = merkki;
                      this.malli = malli;
                      this.vuosimalli = vuosimalli;
                  }

                  public string Merkki
                  {
                      get { return merkki; }
                      set { merkki = value; }
                  }

                  public string Malli
                  {
                      get { return malli; }
                      set { malli = value; }
                  }

                  public int Vuosimalli
                  {
                      get { return vuosimalli; }
                      set { vuosimalli = value; }
                  }

                  public void Aja()
                  {
                      Console.WriteLine("Ajetaan...");
                  }

                  public void Kiihdyta()
                  {
                      Console.WriteLine("Kiihdytetään...");
                  }

                  public void Jarruta()
                  {
                      Console.WriteLine("Jarrutetaan...");
                  }

                  public void Kaynnista()
                  {
                      Console.WriteLine("Käynnistetään...");
                  }

                  public void Sammuta()
                  {
                      Console.WriteLine("Sammutetaan...");
                  }
              }

              class Auto : Ajoneuvo
              {
                  public Auto(string merkki, string malli, int vuosimalli) : base(merkki, malli, vuosimalli) {}
              }

              class TestiAjo
              {

C07. Jatka edellistä tehtävää siten, että lisäät Auto-luokkaan pyorat muuttujan ja oletuskonstruktorissa annat pyorat muuttujalle oletusarvon 4. Tee sitten luokka Moottoripyora, joka perii luokan Auto, mutta pyorat muuttujan oletusarvo on 2.

C#

              ... // Kuten edellä, alla muutokset
              class Auto : Ajoneuvo
              {
                  private int pyorat;
                  public Auto(string merkki, string malli, int vuosimalli) : base(merkki, malli, vuosimalli)
                  {
                      pyorat = 4;
                  }
                  public int Pyorat
                  {
                      get { return pyorat; }
                      set { pyorat = value; }
                  }
              }

              class Moottoripyora : Auto
              {
                  public Moottoripyora(string merkki, string malli, int vuosimalli) : base(merkki, malli, vuosimalli)
                  {
                      Pyorat = 2;
                  }
              }

              class TestiAjo
              {
                  static void Main(string[] args)
                  {
                      Auto mersu = new Auto("Mercedes-Benz", "S-Class", 2020);
                      mersu.Aja();
                      Console.WriteLine(mersu.Pyorat); // tulostaa 4
                      
                      Moottoripyora pyora = new Moottoripyora("Suzuki","Hayabusa",2022);
                      pyora.Aja();
                      Console.WriteLine(pyora.Pyorat); // tulostaa 2
                  }
              }
